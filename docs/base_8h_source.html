459 static inline int ccs_datum_cmp(ccs_datum_t a, ccs_datum_t b) {
460  if (a.type < b.type) {
461  return -1;
462  } else if (a.type > b.type) {
463  return 1;
464  } else {
465  switch(a.type) {
466  case CCS_STRING:
467  if (a.value.s == b.value.s)
468  return 0;
469  else if (!a.value.s)
470  return -1;
471  else if (!b.value.s)
472  return 1;
473  else
474  return strcmp(a.value.s, b.value.s);
475  break;
476  case CCS_INTEGER:
477  return a.value.i < b.value.i ? -1 :
478  a.value.i > b.value.i ? 1 : 0;
479  break;
480  case CCS_FLOAT:
481  return a.value.f < b.value.f ? -1 :
482  a.value.f > b.value.f ? 1 : 0;
483  break;
484  case CCS_NONE:
485  case CCS_INACTIVE:
486  return 0;
487  break;
488  default:
489  return memcmp(&(a.value), &(b.value), sizeof(ccs_value_t));
490  }
491  }
